   - Database: `opencode`
   - Username: `postgres`
   - Password: `postgres`
   - Port: 5432 (use Docker dynamic port mapping via `.WithPortBinding(5432, true)`)

2. After container starts, build connection string from the container's dynamic host/port

3. Apply EF Core migrations for both DragonBall and Music schemas:
   - Use `DragonBallContext` with `context.Database.MigrateAsync()`
   - Use `MusicContext` with `context.Database.MigrateAsync()`
   - Build each with `DbContextOptionsBuilder<T>.UseNpgsql(connectionString)`

4. Expose:
   - `string ConnectionString { get; }` — the connection string for the running container
   - `CancellationToken CancellationToken { get; }` — linked to container's stopping token

5. Implement `InitializeAsync()` and `DisposeAsync()` for container lifecycle

Use the following exact code structure:

```csharp
using Npgsql;
using Microsoft.EntityFrameworkCore;
using OpenCode.Domain.Data;
using Testcontainers.PostgreSql;

namespace OpenCode.Integration.Tests.Fixtures;

public class PostgresFixture : IAsyncLifetime
{
    private readonly PostgreSqlContainer _container;

    public string ConnectionString => _container.GetConnectionString();
    public CancellationToken CancellationToken { get; private set; }

    public PostgresFixture()
    {
        _container = new PostgreSqlBuilder()
            .WithImage("postgres:17")
            .WithDatabase("opencode")
            .WithUsername("postgres")
            .WithPassword("postgres")
            .WithPortBinding(5432, true)
            .WithCleanUp(true)
            .Build();
    }

    public async Task InitializeAsync()
    {
        await _container.StartAsync();
        CancellationToken = _container.GetCancellationToken();

        // Apply DragonBall migrations
        var dragonBallOptions = new DbContextOptionsBuilder<DragonBallContext>()
            .UseNpgsql(ConnectionString)
            .Options;
        using var dragonBallContext = new DragonBallContext(dragonBallOptions);
        await dragonBallContext.Database.MigrateAsync(CancellationToken);

        // Apply Music migrations
        var musicOptions = new DbContextOptionsBuilder<MusicContext>()
            .UseNpgsql(ConnectionString)
            .Options;
        using var musicContext = new MusicContext(musicOptions);
        await musicContext.Database.MigrateAsync(CancellationToken);
    }

    public async Task DisposeAsync()
    {
        await _container.DisposeAsync();
    }
}
```

Note: The `Npgsql` package reference is included in the csproj for `NpgsqlConnection` if needed by tests.
