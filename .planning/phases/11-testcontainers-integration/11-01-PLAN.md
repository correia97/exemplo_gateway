```csharp
using Npgsql;
using Microsoft.EntityFrameworkCore;
using OpenCode.Domain.Data;
using OpenCode.Domain.Entities;
using Testcontainers.PostgreSql;

namespace OpenCode.Integration.Tests.Fixtures;

public class PostgresFixture : IAsyncLifetime
{
    private readonly PostgreSqlContainer _container;

    public string ConnectionString => _container.GetConnectionString();
    public string DragonBallUserConnectionString { get; private set; } = string.Empty;
    public string MusicUserConnectionString { get; private set; } = string.Empty;
    public CancellationToken CancellationToken { get; private set; }

    public PostgresFixture()
    {
        _container = new PostgreSqlBuilder()
            .WithImage("postgres:17")
            .WithDatabase("opencode")
            .WithUsername("postgres")
            .WithPassword("postgres")
            .WithPortBinding(5432, true)
            .WithCleanUp(true)
            .Build();
    }

    public async Task InitializeAsync()
    {
        // Step 1: Start container
        await _container.StartAsync();
        var ct = _container.GetCancellationToken();
        CancellationToken = ct;

        // Step 2: Apply DragonBall migrations
        var dragonBallOptions = new DbContextOptionsBuilder<DragonBallContext>()
            .UseNpgsql(ConnectionString)
            .Options;
        using (var dragonBallContext = new DragonBallContext(dragonBallOptions))
        {
            await dragonBallContext.Database.MigrateAsync(ct);
        }

        // Step 3: Apply Music migrations
        var musicOptions = new DbContextOptionsBuilder<MusicContext>()
            .UseNpgsql(ConnectionString)
            .Options;
        using (var musicContext = new MusicContext(musicOptions))
        {
            await musicContext.Database.MigrateAsync(ct);
        }

        // Step 4: Seed shared baseline dataset (per D-02)
        await SeedBaselineAsync(ct);

        // Step 5: Create test users for schema isolation tests (per D-07)
        await CreateTestUsersAsync(ct);
    }

    private async Task SeedBaselineAsync(CancellationToken ct)
    {
        // --- DragonBall baseline ---
        var dbOptions = new DbContextOptionsBuilder<DragonBallContext>()
            .UseNpgsql(ConnectionString)
            .Options;
        using (var ctx = new DragonBallContext(dbOptions))
        {
            if (!await ctx.Planets.AnyAsync(ct))
            {
                var earth = new Planet { Name = "Earth" };
                ctx.Planets.Add(earth);
                await ctx.SaveChangesAsync(ct);

                var goku = new Character
                {
                    Name = "Goku",
                    Race = "Saiyan",
                    Ki = "10.000.000",
                    MaxKi = "100.000.000.000.000",
                    IntroductionPhase = "Dragon Ball",
                    Description = "Main protagonist",
                    PlanetId = earth.Id
                };
                ctx.Characters.Add(goku);
                await ctx.SaveChangesAsync(ct);
            }
        }

        // --- Music baseline ---
        var musicOptions = new DbContextOptionsBuilder<MusicContext>()
            .UseNpgsql(ConnectionString)
            .Options;
        using (var ctx = new MusicContext(musicOptions))
        {
            if (!await ctx.Genres.AnyAsync(ct))
            {
                ctx.Genres.Add(new Genre { Name = "Rock", Description = "Rock music genre" });
                await ctx.SaveChangesAsync(ct);
            }
        }
    }

    private async Task CreateTestUsersAsync(CancellationToken ct)
    {
        await using var conn = new NpgsqlConnection(ConnectionString);
        await conn.OpenAsync(ct);

        // Create dragonball_user if not exists
        await using (var cmd = conn.CreateCommand())
        {
            cmd.CommandText = """
                DO $$
                BEGIN
                    IF NOT EXISTS (SELECT FROM pg_catalog.pg_roles WHERE rolname = 'dragonball_user') THEN
                        CREATE ROLE dragonball_user WITH LOGIN PASSWORD 'test_password';
                    END IF;
                END
                $$;
                GRANT USAGE ON SCHEMA dragonball TO dragonball_user;
                GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA dragonball TO dragonball_user;
                ALTER DEFAULT PRIVILEGES IN SCHEMA dragonball GRANT SELECT, INSERT, UPDATE, DELETE ON TABLES TO dragonball_user;
                """;
            await cmd.ExecuteNonQueryAsync(ct);
        }

        // Create music_user if not exists
        await using (var cmd = conn.CreateCommand())
        {
            cmd.CommandText = """
                DO $$
                BEGIN
                    IF NOT EXISTS (SELECT FROM pg_catalog.pg_roles WHERE rolname = 'music_user') THEN
                        CREATE ROLE music_user WITH LOGIN PASSWORD 'test_password';
                    END IF;
                END
                $$;
                GRANT USAGE ON SCHEMA music TO music_user;
                GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA music TO music_user;
                ALTER DEFAULT PRIVILEGES IN SCHEMA music GRANT SELECT, INSERT, UPDATE, DELETE ON TABLES TO music_user;
                """;
            await cmd.ExecuteNonQueryAsync(ct);
        }

        // Build user-specific connection strings
        var builder = new NpgsqlConnectionStringBuilder(ConnectionString);
        builder.Username = "dragonball_user";
        builder.Password = "test_password";
        DragonBallUserConnectionString = builder.ConnectionString;

        builder.Username = "music_user";
        builder.Password = "test_password";
        MusicUserConnectionString = builder.ConnectionString;
    }

    public async Task DisposeAsync()
    {
        await _container.DisposeAsync();
    }
}
```

Key design notes:
- `SeedBaselineAsync()` uses `AnyAsync()` checks so it's idempotent (safe if called multiple times)
- `CreateTestUsersAsync()` uses `DO $$ ... $$` block with `IF NOT EXISTS` for idempotent user creation
- User connection strings are built from the base connection string with swapped credentials
- Default privileges ensure tables created by subsequent migrations are also accessible
- The `NpgsqlConnectionStringBuilder` class is used to safely modify connection strings
