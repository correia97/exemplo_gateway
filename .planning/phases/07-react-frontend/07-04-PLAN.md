3. **src/auth/AuthGuard.tsx** — Role-checking wrapper component:
   ```typescript
   import { useAuth } from './AuthProvider'
   
   interface Props {
     roles?: string[]        // Required roles — user must have at least one
     fallback?: React.ReactNode  // What to show if unauthorized (default: nothing)
     children: React.ReactNode
   }
   
   export function AuthGuard({ roles, fallback = null, children }: Props) {
     const { isAuthenticated, hasRole, isLoading } = useAuth()
     if (isLoading) return null
     if (!isAuthenticated) return <>{fallback}</>
     if (roles && roles.length > 0 && !roles.some(r => hasRole(r))) return <>{fallback}</>
     return <>{children}</>
   }
   
   /**
    * Convenience: checks if user has "write" role (required for POST/PUT/DELETE)
    */
   export function WriteGuard({ children, fallback }: Omit<Props, 'roles'>) {
     return <AuthGuard roles={['write', 'admin', 'opencode-admin']} fallback={fallback}>{children}</AuthGuard>
   }
   ```

1. **src/components/ErrorBoundary.tsx** — React error boundary class component:
   ```typescript
   import { Component, type ErrorInfo, type ReactNode } from 'react'
   
   interface Props { children: ReactNode; fallback?: ReactNode }
   interface State { hasError: boolean; error: Error | null }
   
   export default class ErrorBoundary extends Component<Props, State> {
     constructor(props: Props) {
       super(props)
       this.state = { hasError: false, error: null }
     }
     static getDerivedStateFromError(error: Error): State {
       return { hasError: true, error }
     }
     componentDidCatch(error: Error, info: ErrorInfo) {
       console.error('ErrorBoundary caught:', error, info.componentStack)
     }
     handleReset = () => {
       this.setState({ hasError: false, error: null })
     }
     render() {
       if (this.state.hasError) {
         if (this.props.fallback) return this.props.fallback
         return (
           <div style={{ padding: '2rem', textAlign: 'center' }}>
             <h2>Something went wrong</h2>
             <pre style={{ color: '#c00', margin: '1rem 0', whiteSpace: 'pre-wrap' }}>
               {this.state.error?.message}
             </pre>
             <button onClick={this.handleReset} style={{ padding: '0.5rem 1.5rem', cursor: 'pointer' }}>
               Try Again
             </button>
           </div>
         )
       }
       return this.props.children
     }
   }
   ```

3. **src/hooks/useApiError.ts** — Hook for consistent error state management:
   ```typescript
   import { useState, useCallback } from 'react'
   import { ApiClientError } from '../api/client'
   
   interface ApiErrorState {
     message: string | null
     correlationId: string | null
   }
   
   export function useApiError() {
     const [error, setError] = useState<ApiErrorState>({ message: null, correlationId: null })
   
     const handleError = useCallback((err: unknown) => {
       if (err instanceof ApiClientError) {
         setError({ message: err.message, correlationId: err.correlationId ?? null })
       } else if (err instanceof Error) {
         setError({ message: err.message, correlationId: null })
       } else {
         setError({ message: 'An unknown error occurred', correlationId: null })
       }
     }, [])
   
     const dismissError = useCallback(() => {
       setError({ message: null, correlationId: null })
     }, [])
   
     return { error, handleError, dismissError }
   }
   ```
