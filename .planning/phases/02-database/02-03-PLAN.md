    ```csharp
    namespace OpenCode.Domain.Pagination;

    /// <summary>
    /// Generic pagination envelope for list endpoints.
    /// All list endpoints return this wrapper to provide consistent pagination metadata.
    /// </summary>
    public class PagedResult<T>
    {
        /// <summary>
        /// The paginated items for the current page.
        /// </summary>
        public IEnumerable<T> Data { get; set; } = Enumerable.Empty<T>();

        /// <summary>
        /// Total number of items across all pages.
        /// </summary>
        public int TotalCount { get; set; }

        /// <summary>
        /// Current page number (1-based).
        /// </summary>
        public int Page { get; set; }

        /// <summary>
        /// Number of items per page.
        /// </summary>
        public int PageSize { get; set; }

        /// <summary>
        /// Total number of pages, computed from TotalCount and PageSize.
        /// </summary>
        public int TotalPages => (int)Math.Ceiling(TotalCount / (double)PageSize);
    }
    ```

    Create `src/OpenCode.Domain/Interfaces/IRepository.cs`:

    ```csharp
    using OpenCode.Domain.Pagination;

    namespace OpenCode.Domain.Interfaces;

    /// <summary>
    /// Generic repository interface for CRUD operations with pagination.
    /// All entity-specific repositories inherit from this interface.
    /// </summary>
    public interface IRepository<T> where T : class
    {
        /// <summary>
        /// Gets an entity by its primary key.
        /// </summary>
        Task<T?> GetByIdAsync(int id, CancellationToken cancellationToken = default);

        /// <summary>
        /// Gets a paginated list of all entities.
        /// Default page size is 10 (standard for this project).
        /// </summary>
        Task<PagedResult<T>> GetAllAsync(
            int page = 1,
            int pageSize = 10,
            CancellationToken cancellationToken = default);

        /// <summary>
        /// Adds a new entity to the repository.
        /// Returns the entity with its generated Id populated.
        /// </summary>
        Task<T> AddAsync(T entity, CancellationToken cancellationToken = default);

        /// <summary>
        /// Updates an existing entity.
        /// </summary>
        Task UpdateAsync(T entity, CancellationToken cancellationToken = default);

        /// <summary>
        /// Deletes an entity from the repository.
        /// </summary>
        Task DeleteAsync(T entity, CancellationToken cancellationToken = default);
    }
    ```
